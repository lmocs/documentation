# Add fonts using Nerd Fonts
cd /usr/share/fonts  
sudo mkdir font-name  
cd font-name  
sudo curl -OL https://github.com/ryanoasis/nerd-fonts/releases/latest/download/font-name.tar.xz  
sudo tar -xf font-name.tar.xz  
fc-list | grep font-name
